import numpy as np

X = np.array([[2,9],[3,6],[1,5]])
y = np.array([[89],[96],[92]])

# Normalize data
X = X / np.max(X)
y = y / 100

# Sigmoid function
def sigmoid(x):
    return 1 / (1 + np.exp(-x))

# Derivative of sigmoid
def sigmoid_prime(x):
    return x * (1 - x)

# Initialize weights
w1 = np.random.rand(2,3)
w2 = np.random.rand(3,1)

# Output before training
hidden = sigmoid(np.dot(X, w1))
output = sigmoid(np.dot(hidden, w2))

print("Input:\n", X)
print("Predicted Output:\n", output)
print("Loss Before Training:\n",
      np.mean(np.square(y - output)))

# Training
for i in range(1000):
    hidden = sigmoid(np.dot(X, w1))
    output = sigmoid(np.dot(hidden, w2))

    error = y - output
    d_output = error * sigmoid_prime(output)
    d_hidden = np.dot(d_output, w2.T) * sigmoid_prime(hidden)

    w2 += np.dot(hidden.T, d_output)
    w1 += np.dot(X.T, d_hidden)

# Output after training
hidden = sigmoid(np.dot(X, w1))
output = sigmoid(np.dot(hidden, w2))

print("Predicted Output:\n", output)
print("Loss After Training:\n",
      np.mean(np.square(y - output)))

import matplotlib.pyplot as plt
import numpy as np

X = np.array([1,2,3,4,5])
y = np.array([1,2,1.2,2.3,3.2])

x_mat = np.c_[np.ones(len(X)), X]

def weights(X_query, X, tau):
    return np.exp(-((X - X_query)**2) / (2 * tau**2))

def lwr(X_query, X, y, tau):
    wt = weights(X_query, X, tau)
    W = np.diag(wt)
    theta = np.linalg.pinv(x_mat.T @ W @ x_mat) @ (x_mat.T @ W @ y)
    y_pred = theta[0] + theta[1] * X_query
    return y_pred

X_query = 3
tau = 1.0

y_pred = lwr(X_query, X, y, tau)

x_val = np.linspace(1, 5, 100)
y_val = [lwr(x, X, y, tau) for x in x_val]

plt.plot(x_val, y_val, color='blue', label='LWR')
plt.scatter(X, y, color='red', label='Data points')
plt.scatter(X_query, y_pred, color='green', label='Query point')
plt.legend()
plt.show()

import pandas as pd
import math

data = pd.read_csv("tennis.csv")
target = data.columns[-1]

def entropy(col):
    e = 0
    for v in col.unique():
        p = sum(col == v) / len(col)
        e -= p * math.log2(p)
    return e

def id3(data, attrs):
    y = data[target]

    if len(y.unique()) == 1:
        return y.iloc[0]

    gains = {}
    for a in attrs:
        g = entropy(y)
        for v in data[a].unique():
            s = data[data[a] == v]
            g -= len(s)/len(data) * entropy(s[target])
        gains[a] = g

    best = max(gains, key=gains.get)
    tree = {best: {}}

    for v in data[best].unique():
        s = data[data[best] == v]
        if len(s) == 0:
            tree[best][v] = y.mode()[0]
        else:
            tree[best][v] = id3(
                s.drop(columns=[best]),
                [x for x in attrs if x != best]
            )
    return tree

def print_tree(tree, space=""):
    if not isinstance(tree, dict):
        print(space + str(tree))
        return

    for k, v in tree.items():
        print(space + k)
        for val, child in v.items():
            print(space + "   " + str(val))
            print_tree(child, space + "      ")

tree = id3(data, list(data.columns[:-1]))
print_tree(tree)
